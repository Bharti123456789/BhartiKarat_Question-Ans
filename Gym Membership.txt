     * of minutes spent from the start of the day.

    public Map<Integer, Double> getAverageWorkoutDurations() {

        for (Member member : members) {

            if (member.workout == null || member.workout.isEmpty()) {
                continue;
            }

            int totalDuration = 0;
            for (Workout w : member.workout) {
                totalDuration += w.getDuration();
            }

            double avg = totalDuration / (double) member.workout.size();
            averageDurations.put(member.memberId, avg);
        }

        return averageDurations;
    }
